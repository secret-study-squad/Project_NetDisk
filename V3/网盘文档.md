# 用户表

```sql
create table user(id int primary key auto_increment, user_name varchar(10) not NULL unique, salt varchar(32) not NULL, passwd_hash varchar (64) not NULL);
```

# 哈希表

```sql
create table file_forest (
  id int primary key AUTO_INCREMENT,
  file_name varchar(255) NOT NULL,
  user_name varchar(20) NOT NULL,
  pdir_id int NOT NULL,
  path varchar(1024) NOT NULL,
  file_type int NOT NULL,
  file_hash varchar(64) DEFAULT NULL,
  create_time datetime DEFAULT CURRENT_TIMESTAMP,
  alive int DEFAULT 0,
  file_size bigint DEFAULT 0,
  is_complete tinyint DEFAULT 0 COMMENT '0-不完整 1-完整'
) ENGINE = InnoDB DEFAULT CHARSET = utf8mb4 COLLATE = utf8mb4_0900_ai_ci ROW_FORMAT = Dynamic;
```

# 哈希文件表

```sql
create table file_hash (
    id int primary key auto_increment,
    file_size bigint not NULL,
    file_hash VARCHAR(64) not NULL unique
);
```
